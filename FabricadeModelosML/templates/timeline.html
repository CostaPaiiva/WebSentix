@app.route("/timeline/") def timeline(projeto): import os, json pasta_projeto = os.path.join(PROJECTS_FOLDER, projeto) if not os.path.exists(pasta_projeto): return f"Projeto {projeto} não encontrado", 404 historico = [] for versao in os.listdir(pasta_projeto): pasta_versao = os.path.join(pasta_projeto, versao) if not os.path.isdir(pasta_versao): continue caminho_meta = os.path.join(pasta_versao, "meta.json") if not os.path.exists(caminho_meta): continue try: with open(caminho_meta, "r", encoding="utf-8") as f: meta = json.load(f) # Garante que o campo comentário sempre exista if "comentario" not in meta: meta["comentario"] = "" historico.append(meta) except Exception as e: print("❌ Erro ao ler meta.json de", versao, ":", e) # Ordena por data (mais recente primeiro) historico = sorted(historico, key=lambda x: x.get("data", ""), reverse=True) # Descobre qual está em produção producao = None caminho_prod = os.path.join(pasta_projeto, "PRODUCAO.txt") if os.path.exists(caminho_prod): with open(caminho_prod, "r", encoding="utf-8") as f: producao = f.read().strip() # 🔥 ESSE RETURN É O QUE ESTAVA FALTANDO EM ALGUM CAMINHO return render_template( "timeline.html", projeto=projeto, historico=historico, producao=producao )
